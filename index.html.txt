        // 🎵 HTML5 原生音樂相對路徑控制
        const bgMusic = document.getElementById('bgMusic');
        const musicBtn = document.getElementById('musicBtn');
        let isMusicPlaying = false;

        musicBtn.addEventListener('click', function() {
            if (isMusicPlaying) {
                bgMusic.pause();
                musicBtn.classList.remove('playing');
            } else {
                bgMusic.play().then(() => {
                    musicBtn.classList.add('playing');
                }).catch(err => {
                    console.log("播放被瀏覽器安全政策阻擋，請再試一次：", err);
                });
            }
            isMusicPlaying = !isMusicPlaying;

        // 智能表單邏輯：根據參加人數判斷是否顯示飲食需求與兒童椅
        function handleAttendeesChange(val) {
            const count = parseInt(val, 10);
            toggleConditionalFields(count);
        }

        function toggleConditionalFields(count) {
            const dietGroup = document.getElementById('dietGroup');
            const chairGroup = document.getElementById('chairGroup');
            const dietSelect = document.getElementById('dietSelect');
            const chairSelect = document.getElementById('chairSelect');

            if (count > 0) {
                // 顯示隱藏欄位並設為必填
                dietGroup.style.display = 'block';
                chairGroup.style.display = 'block';
                dietSelect.setAttribute('required', 'required');
                chairSelect.setAttribute('required', 'required');
            } else {
                // 隱藏欄位並取消必填、重設數值
                dietGroup.style.display = 'none';
                chairGroup.style.display = 'none';
                dietSelect.removeAttribute('required');
                chairSelect.removeAttribute('required');
                dietSelect.value = "";
                chairSelect.value = "";
            }
        }

        // 倒數計時器邏輯：設定目標時間 2026/11/08 17:30 (使用 ISO 8601 格式，100% 相容 iOS Safari 系統)
        const targetDate = new Date('2026-11-08T17:30:00').getTime();

        function updateCountdown() {
            const now = new Date().getTime();
            const distance = targetDate - now;

            if (distance < 0) {
                document.getElementById('days').innerText = "00";
                document.getElementById('hours').innerText = "00";
                document.getElementById('minutes').innerText = "00";
                document.getElementById('seconds').innerText = "00";
                clearInterval(countdownInterval);
                return;
            }

            const d = Math.floor(distance / (1000 * 60 * 60 * 24));
            const h = Math.floor((distance % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
            const m = Math.floor((distance % (1000 * 60 * 60)) / (1000 * 60));
            const s = Math.floor((distance % (1000 * 60)) / 1000);

            // 補零顯示
            document.getElementById('days').innerText = d < 10 ? '0' + d : d;
            document.getElementById('hours').innerText = h < 10 ? '0' + h : h;
            document.getElementById('minutes').innerText = m < 10 ? '0' + m : m;
            document.getElementById('seconds').innerText = s < 10 ? '0' + s : s;
        }

        // 初始化與每秒執行
        updateCountdown();
        const countdownInterval = setInterval(updateCountdown, 1000);

        // 表單提交互動與 Modal 彈窗控制

                rsvpForm.reset();
                toggleConditionalFields(0); // 隱藏動態欄位並重設
            }, 300);
        });

        closeModalBtn.addEventListener('click', function() {
            successModal.classList.remove('active');
        });
